{{- $u := urls.Parse .Destination -}} {{- $src := $u.String -}} {{- /* 优先从 Page Resources（Page Bundle）中获取图片 */ -}} {{- $resource := "" -}} {{- if not $u.IsAbs -}} {{- $path := strings.TrimPrefix "./" $u.Path -}} {{- $resource = or (.PageInner.Resources.Get $path) (.Page.Resources.Get $path) (.Page.Resources.GetMatch $path) -}} {{- end -}} {{- /* 如果找到了 Page Resource，则使用它的 RelPermalink，否则 fallback 到静态资源 */ -}} {{- if $resource -}} {{- $src = $resource.RelPermalink -}} {{- else if not $u.IsAbs -}} {{- $path := strings.TrimPrefix "./" $u.Path -}} {{- with resources.Get $path -}} {{- $src = .RelPermalink -}} {{- end -}} {{- end -}} {{- /* 处理查询参数和锚点 */ -}} {{- with $u.RawQuery -}} {{- $src = printf "%s?%s" $src . -}} {{- end -}} {{- with $u.Fragment -}} {{- $src = printf "%s#%s" $src . -}} {{- end -}} {{- $attributes := merge .Attributes (dict "alt" .Text "src" $src "title" (.Title | transform.HTMLEscape) "loading" "lazy" ) -}}
